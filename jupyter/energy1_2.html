# prepare the functions for evaluating the model
def mse(ar1, ar2):
    return ((ar1 - ar2) ** 2).mean()

def rmse(ar1, ar2):
    return np.sqrt(mse(ar1, ar2))

# Initialising the RNN
layers = Sequential()

layers.add(Flatten())

# Adding the output layer
